\begin{algorithm}[t]
\caption{(Non-contextual case) Discounted UCB}
\label{alg:ducb}
\textbf{Require:} Finite action set $\mathcal A$; discount factor $\gamma\in(0,1)$; exploration constant $\alpha>0$.
\begin{algorithmic}[1]
\Procedure{\textsf{Initialize}}{}
    \State Initialize statistics $\forall{\tau\in\mathcal A},\;\{N[\tau]\gets 0,\; S[\tau]\gets 0\}$ and action cache $C\gets \texttt{null}$.
    \State \Return $\pi^{(0)} \triangleq \{N,S,C\}$.
\EndProcedure
\Procedure{\textsf{Pull}}{$t,\pi$}
    \If{$\exists \tau\in\mathcal A$ s.t. $N[\tau]=0$}
        \State $\tau^{(t)}\gets$ any such $\tau$; increase $N[\tau]=1$.
    \Else
        \State $\hat r_\tau\gets S[\tau]/N[\tau]$ for all $\tau\in\mathcal A$.
        \State $\tau^{(t)} \gets \arg\max_{\tau\in\mathcal A}\left\{\hat r_\tau+\alpha\sqrt{\frac{\log t}{N[\tau]}}\right\}$.
    \EndIf
    \State $C\gets \tau^{(t)}$; \Return $\tau^{(t)}$.
\EndProcedure
\Procedure{\textsf{Adapt}}{$r,\pi$}
    \If{$C=\texttt{null}$} \State \Return $\pi$. \EndIf
    \State $\tau \gets C$ and $C\gets \texttt{null}$.
    \ForAll{$a\in\mathcal A$} \State $N[a]\gets \gamma N[a]$; \;\; $S[a]\gets \gamma S[a]$. \EndFor
    \State $N[\tau]\gets N[\tau]+1$; \;$S[\tau]\gets S[\tau]+r$.
    \State \Return $\pi$.
\EndProcedure
\end{algorithmic}
\end{algorithm}
