\begin{algorithm}[t]
\caption{(Contextual case; per-client) Discounted LinUCB}
\label{alg:dlinucb}
\textbf{Require:} Finite action set $\mathcal A$; feature map $\varphi(\boldsymbol{x},\tau)\in\mathbb R^d$; discount factor $\gamma\in(0,1)$; ridge parameter $\alpha>0$; exploration parameter $\beta>0$.
\begin{algorithmic}[1]
\Procedure{\textsf{Initialize}}{}
    \State Initialize $\boldsymbol{A}\gets \alpha \boldsymbol{I}_d$, $\boldsymbol{b}\gets \boldsymbol{0}_d$, and feature cache $C\gets \texttt{null}$.
    \State \Return $\pi^{(0)} \triangleq \{\boldsymbol{A},\boldsymbol{b},C\}$.
\EndProcedure
\Procedure{\textsf{Pull}}{$t,\pi,\{\boldsymbol{x}_i^{(t-1)}\}_{i=1}^K$}
    \State $\hat{\boldsymbol{w}} \gets \boldsymbol{A}^{-1}\boldsymbol{b}$.
    \For{$i=1$ to $K$}
        \State $\tau_i^{(t)} \gets \arg\max_{\tau\in\mathcal A}\left\{\langle \varphi(\boldsymbol{x}_i^{(t-1)},\tau),\hat{\boldsymbol{w}}\rangle
        + \beta \big\|\varphi(\boldsymbol{x}_i^{(t-1)},\tau)\big\|_{\boldsymbol{A}^{-1}}\right\}$.
    \EndFor
    \State $C \gets \psi^{(t)}$ where $\psi^{(t)} \gets \sum_{i=1}^K \varphi(\boldsymbol{x}_i^{(t-1)},\tau_i^{(t)})$.
    \State \Return $\tau^{(t)} = [\tau_1^{(t)},\ldots,\tau_K^{(t)}]^\top$.
\EndProcedure
\Procedure{\textsf{Adapt}}{$r,\pi$}
    \If{$C=\texttt{null}$} \State \Return $\pi$. \EndIf
    \State $\psi \gets C$ and $C\gets \texttt{null}$.
    \State $\boldsymbol{A} \gets \gamma \boldsymbol{A} + \psi\psi^\top + (1-\gamma)\alpha \boldsymbol{I}_d$.
    \State $\boldsymbol{b} \gets \gamma \boldsymbol{b} + r\psi$.
    \State \Return $\pi$.
\EndProcedure
\end{algorithmic}
\end{algorithm}
