\begin{algorithm}[t]
\caption{(Non-contextual case) Discounted TS}
\label{alg:dts}
\textbf{Require:} Finite action set $\mathcal A$; discount factor $\gamma\in(0,1)$; Gaussian likelihood variance $\alpha>0$; prior variance $\beta>0$.
\begin{algorithmic}[1]
\Procedure{\textsf{Initialize}}{}
    \State Initialize statistics $\forall{\tau\in\mathcal A},\;\{N[\tau]\gets 0,\; S[\tau]\gets 0\}$ and action cache $C\gets \texttt{null}$.
    \State \Return $\pi^{(0)} \triangleq \{N,S,C\}$.
\EndProcedure
\Procedure{\textsf{Pull}}{$t,\pi$}
    \ForAll{$\tau\in\mathcal A$}
        \State $v_\tau \gets \left(\beta^{-1}+\alpha^{-1}N[\tau]\right)^{-1}$.
        \State $m_\tau \gets \alpha^{-1} v_\tau \cdot S[\tau]$.
        \State Sample $\hat{r}_\tau \sim \mathcal N(m_\tau,v_\tau)$.
    \EndFor
    \State $\tau^{(t)} \gets \arg\max_{\tau\in\mathcal A}\hat{r}_\tau$.
    \State $C\gets \tau^{(t)}$; \Return $\tau^{(t)}$.
\EndProcedure
\Procedure{\textsf{Adapt}}{$r,\pi$}
    \If{$C=\texttt{null}$} \State \Return $\pi$. \EndIf
    \State $\tau \gets C$ and $C\gets \texttt{null}$.
    \ForAll{$a\in\mathcal A$} \State $N[a]\gets \gamma N[a]$; \;\; $S[a]\gets \gamma S[a]$. \EndFor
    \State $N[\tau]\gets N[\tau]+1$; \;\; $S[\tau]\gets S[\tau]+r$.
    \State \Return $\pi$.
\EndProcedure
\end{algorithmic}
\end{algorithm}
